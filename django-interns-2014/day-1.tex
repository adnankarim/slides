\section{Django project}
\label{sec-3}
\begin{frame}[fragile]
\frametitle{Django project}
\label{sec-3-1}
\begin{block}{Create a project}
\label{sec-3-1-1}


\begin{minted}[]{sh}
django-admin startproject <PROJECT_NAME>
\end{minted}
\end{block}
\end{frame}
\begin{frame}[fragile]
\frametitle{Django app}
\label{sec-3-2}
\begin{block}{Create an app}
\label{sec-3-2-1}


\begin{minted}[]{sh}
cd <PROJECT DIR>
python manage.py startapp <APP_NAME>
\end{minted}
\end{block}
\end{frame}
\begin{frame}[fragile]
\frametitle{Django server (localhost:8000)}
\label{sec-3-3}

\label{sec-3-3-1}


\begin{minted}[]{sh}
cd <PROJECT DIR>
python manage.py runserver
\end{minted}
\end{block}
\end{frame}
